200 template<class InputIterator, class OutputIterator>
201 OutputIterator Plane::ClipPolygon(InputIterator inputBegin, InputIterator inputEnd, OutputIterator outBegin)
202 {
203  assert(inputEnd - inputBegin <= 128);
204  std::array<Vector3, 128> intermediatePoints;
205  auto interEnd = std::copy(inputBegin, inputEnd, intermediatePoints.begin());
206  auto interBegin = intermediatePoints.begin();
207 
208  // Find the first vertex that lies on the front of the plane.
209  auto start = interEnd;
210  for(auto i = interBegin; i != interEnd; i++ )
211  {
212  if(Check(*i) == kInside)
213  {
214  start = i;
215  break;
216  }
217  }
218 
219  // If nothing was in front of the plane, we're done.
220  if(start == interEnd)
221  return outBegin;
222 
223  auto baseStart = start;
224  auto end = start;
225  IncrementAndWrap(end, interBegin, interEnd);
226 
227  auto outIterator = outBegin;
228  while(end != baseStart)
229  {
230  IntersectStatus startSide = Check(*start);
231  IntersectStatus endSide = Check(*end);
232 
233  *outIterator++ = *start;
234  if(startSide != kOutside && endSide != kOutside)
235  {
236  // No clipping required
237  start = end;
238  IncrementAndWrap(end, interBegin, interEnd);
239  }
240  else if(startSide != kOutside && endSide == kOutside)
241  {
242  if(startSide == kInside) // Insert the intersection:
243  *outIterator++ = IntersectionPoint(*start, *end);
244 
245  // Advance to first point that is in front or on, use as next end point:
246  do
247  {
248  IncrementAndWrap(end, interBegin, interEnd);
249  }while(Check(*end) == kOutside);
250 
251  // Step back by one to find start point:
252  start = end;
253  DecrementAndWrap(start, interBegin, interEnd);
254 
255  *start = IntersectionPoint(*start, *end);
256  }
257  else
258  {
259  std::ostringstream oss;
260  oss << "Bad combination: " << startSide << ", " << endSide << "; ";
261  oss << "Plane: " << *this << "; Polygon: ";
262  for(auto it = inputBegin; it != inputEnd; ++it)
263  oss << *it << ", ";
264  throw std::runtime_error(oss.str());
265  }
266  }
267 
268  // Emit the last point.
269  *outIterator++ = *start;
270 
271  return outIterator;
272 }
