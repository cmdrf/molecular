153 inline double AxisAlignedBox::Overlap(const AxisAlignedBox& box) const
154 {
155  if(!Overlaps(box))
156  return 0.0;
157  double d0 = std::min(box.GetMax(0), mMax[0]) - std::max(box.GetMin(0), mMin[0]);
158  double d1 = std::min(box.GetMax(1), mMax[1]) - std::max(box.GetMin(1), mMin[1]);
159  double d2 = std::min(box.GetMax(2), mMax[2]) - std::max(box.GetMin(2), mMin[2]);
160  return d0 * d1 * d2;
161 }
