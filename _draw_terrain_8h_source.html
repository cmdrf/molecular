109  // Build vertex data:
110  std::vector<uint16_t> patchVertexData(kPatchSize * kPatchSize * 2);
111  for(uint16_t y = 0; y < kPatchSize; ++y)
112  {
113  for(uint16_t x = 0; x < kPatchSize; ++x)
114  {
115  patchVertexData[(y * kPatchSize + x) * 2] = x;
116  patchVertexData[(y * kPatchSize + x) * 2 + 1] = y;
117  }
118  }
119  mPatchVertices->Store(&patchVertexData.front(), patchVertexData.size() * sizeof(uint16_t));
120 
121  // Build index data:
122  std::vector<uint16_t> indices;
123  for(int i = 0; i < kLodLevels; ++i)
124  {
125  mPatchIndices[i] = mRenderer.CreateIndexBuffer();
126  BuildIndices(i, indices, mPatchInfos[i]);
127  mPatchIndices[i]->Store(&indices.front(), indices.size() * 2);
128  }
