66 template<class TetrahedronIterator, class PositionIterator, class CoeffsIterator, class NormalsIterator, class Storage>
67 void WriteTetrahedronSpaceFile(
68  TetrahedronIterator tetBegin, TetrahedronIterator tetEnd,
69  PositionIterator posBegin, PositionIterator posEnd,
70  CoeffsIterator coeffsBegin, CoeffsIterator coeffsEnd,
71  NormalsIterator normalsBegin, NormalsIterator normalsEnd,
72  Storage& outStorage)
73 {
74  TetrahedronSpaceFile header;
75  header.magic = TetrahedronSpaceFile::kMagic;
76  header.version = TetrahedronSpaceFile::kVersion;
77  header.numPoints = posEnd - posBegin;
78  assert(header.numPoints == coeffsEnd - coeffsBegin);
79  assert(header.numPoints == normalsEnd - normalsBegin);
80  header.numTetrahedrons = tetEnd - tetBegin;
81  header.verticesOffset = sizeof(TetrahedronSpaceFile) + header.numTetrahedrons * sizeof(TetrahedronSpaceFile::Tetrahedron);
82  header.normalsOffset = header.verticesOffset + header.numPoints * sizeof(TetrahedronSpaceFile::Vertex);
83 
84  outStorage.Write(&header, sizeof(TetrahedronSpaceFile));
85  for(auto it = tetBegin; it != tetEnd; ++it)
86  outStorage.Write(&*it, sizeof(TetrahedronSpaceFile::Tetrahedron));
87  auto pIt = posBegin;
88  auto cIt = coeffsBegin;
89  for(; pIt != posEnd && cIt != coeffsEnd; ++pIt, ++cIt)
90  {
91  TetrahedronSpaceFile::Vertex vertex{*pIt, *cIt};
92  outStorage.Write(&vertex, sizeof(TetrahedronSpaceFile::Vertex));
93  }
94 
95  for(auto it = normalsBegin; it != normalsEnd; ++it)
96  outStorage.Write(&*it, sizeof(util::Vector3));
97 }
98 
99 }
100 }
