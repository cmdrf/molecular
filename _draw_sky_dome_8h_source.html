83 template<class TRenderManager>
84 void DrawSkyDome<TRenderManager>::Execute()
85 {
86  if(!mVertexBuffer)
87  {
88  const unsigned int polarSteps = 8;
89  const unsigned int azimuthSteps = 24;
90  std::vector<float> vertexData;
91  vertexData.reserve(polarSteps * azimuthSteps * 5);
92  const float polarStep = 0.5f * Math::kPi_f / (polarSteps - 1);
93  const float azimuthStep = 2.0f * Math::kPi_f / (azimuthSteps - 1);
94  const float radius = 100;
95  const float uStep = 1.0f / (azimuthSteps - 1);
96  const float vStep = 1.0f / (polarSteps - 1);
97  for(unsigned int p = 0; p < polarSteps; p++)
98  {
99  for(unsigned int a = 0; a < azimuthSteps; a++)
100  {
101  vertexData.push_back(radius * std::sin(p * polarStep) * std::cos(a * azimuthStep)); // X
102  vertexData.push_back(radius * std::sin(p * polarStep) * std::sin(a * azimuthStep)); // Y
103  vertexData.push_back(radius * std::cos(p * polarStep)); // Z
104  vertexData.push_back(uStep * a); // U
105  vertexData.push_back(vStep * p); // V
106  }
107  }
108  mVertexBuffer = mRenderer.CreateVertexBuffer();
109  mVertexBuffer->Store(vertexData.data(), 5 * vertexData.size());
110  mPositionInfo.components = 3;
111  mPositionInfo.stride = 5 * sizeof(float);
112  mPositionInfo.type = VertexAttributeInfo::kFloat;
113  mUvInfo.components = 2;
114  mUvInfo.offset = 3 * sizeof(float);
115  mUvInfo.stride = 5 * sizeof(float);
116  mUvInfo.type = VertexAttributeInfo::kFloat;
117 
118  // Indices:
119  std::vector<uint16_t> indexData;
120  indexData.reserve(polarSteps * azimuthSteps * 6);
121  for(unsigned int p = 0; p < polarSteps - 1; p++)
122  {
123  for(unsigned int a = 0; a < azimuthSteps - 1; a++)
124  {
125  uint16_t p0 = p * azimuthSteps + a;
126  uint16_t p1 = p * azimuthSteps + (a + 1);
127  uint16_t p2 = (p + 1) * azimuthSteps + (a + 1);
128  uint16_t p3 = (p + 1) * azimuthSteps + a;
129  indexData.push_back(p0);
130  indexData.push_back(p1);
131  indexData.push_back(p2);
132  indexData.push_back(p0);
133  indexData.push_back(p2);
134  indexData.push_back(p3);
135  }
136  }
137  mIndexBuffer = mRenderer.CreateIndexBuffer();
138  mIndexBuffer->Store(indexData.data(), indexData.size() * sizeof(uint16_t));
139  mIndexInfo.count = indexData.size();
140  mIndexInfo.type = IndexBufferInfo::Type::kUInt16;
141  }
142 
143  Binding<Uniform<Matrix4>> viewMatrix("viewMatrix"_H, this);
144  Binding<Uniform<Matrix4>> modelMatrix("modelMatrix"_H, this);
145  **viewMatrix = Matrix4::Translation(-(**viewMatrix).GetTranslation()) * **viewMatrix;
146  **modelMatrix = Matrix4::Identity();
147 
148  /* Choose different name for attribute to select program snippet writing 1.0 to Z. */
149  Binding<Attribute> vertexPositionAttr("skyVertexPositionAttr"_H, this);
150  Binding<Attribute> vertexUv0Attr("vertexUv0Attr"_H, this);
151  Binding<Uniform<RenderCmdSink::Texture*>> diffuseTexture("diffuseTexture"_H, this);
152 
153  *vertexPositionAttr = Attribute(mVertexBuffer, mPositionInfo);
154  *vertexUv0Attr = Attribute(mVertexBuffer, mUvInfo);
155  if(mTexture)
156  **diffuseTexture = mTexture->Use();
157  mRenderer.SetDepthState(true, false, RenderCmdSink::kEqual);
158  PrepareProgram();
159  mRenderer.Draw(mIndexBuffer, mIndexInfo);
160  mRenderer.SetDepthState(true, true);
161 
162 }
