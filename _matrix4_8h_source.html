177 inline Matrix4 Matrix4::ProjectionPerspective(float fieldOfView, float aspectRatio, float nearPlane, float farPlane)
178 {
179  Matrix4 outMatrix = Identity();
180  float f = 1.0f / std::tan(fieldOfView / 2.0f);
181  outMatrix(0, 0) = f / aspectRatio;
182  outMatrix(1, 1) = f;
183  outMatrix(2, 2) = (farPlane + nearPlane) / (nearPlane - farPlane);
184  outMatrix(2, 3) = (2.0f * farPlane * nearPlane) / (nearPlane - farPlane);
185  outMatrix(3, 2) = -1.0f;
186  outMatrix(3, 3) = 0.0f;
187  return outMatrix;
188 }
