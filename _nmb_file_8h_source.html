149 template<class Storage>
150 NmbFile::NmbFile(Storage& storage)
151 {
152  LittleEndianReadStream<Storage> stream(storage);
153  uint32_t magic = 0, version1 = 0, version2 = 0;
154  stream.Read(magic);
155 
156  if(magic != 1)
157  throw std::runtime_error("Wrong magic number");
158 
159  stream.Read(version1);
160  stream.Read(version2);
161 
162  while(!storage.EndOfData())
163  {
164  uint32_t marker = 0;
165  stream.Read(marker);
166  if(marker == kEndMarker)
167  break;
168 
169  if((marker & 0xff000000) == 0)
170  continue;
171 
172  ReadMarker(storage, kBeginMarker);
173 
174  std::string name;
175  ReadString(storage, name);
176 // std::cerr << "name: " << name << std::endl;
177 
178  switch(marker)
179  {
180  case kPolySurfaceShape:
181  mMeshes.push_back(Mesh());
182  mMeshes.back().name = name;
183  ReadPolySurfaceShape(storage, mMeshes.back());
184  break;
185 
186  case kProgram:
187  {
188  Program program;
189  ReadProgram(storage, program);
190  }
191  break;
192 
193  default:
194  throw std::runtime_error("Unknown chunk \"" + name + "\"");
195  }
196  }
197 }

199 template<class Storage>
200 void NmbFile::ReadPolySurfaceShape(Storage& storage, Mesh &mesh)
201 {
202  LittleEndianReadStream<Storage> stream(storage);
203  const int elementSizes[8] = {1, 2, 3, 4, 2, 3, 0, 0};
204 
205  uint8_t null = 0;
206  stream.Read(null);
207  uint32_t unknown1 = 0, unknown2 = 0;
208  uint32_t a2vCount = 0;
209  stream.Read(unknown1);
210  stream.Read(unknown2);
211  stream.Read(a2vCount);
212 
213  mesh.vertexBuffers.resize(a2vCount);
214  for(unsigned int i = 0; i < a2vCount; ++i)
215  {
216  VertexBuffer& vertBuffer = mesh.vertexBuffers[i];
217 
218  uint16_t a2vType = 0, unknown3a = 0;
219  uint32_t unknown3b = 0;
220  stream.Read(a2vType);
221  stream.Read(unknown3a);
222  stream.Read(unknown3b);
223 
224  std::string name;
225  ReadString(storage, name);
226 
227  uint32_t unknown4 = 0, unknown5 = 0, unknown6 = 0;
228  uint32_t elementCount = 0;
229  stream.Read(unknown4);
230  stream.Read(unknown5);
231  stream.Read(unknown6);
232  stream.Read(elementCount);
233 
234  unsigned int elementSize = elementSizes[a2vType & 0x7];
235  if(elementSize == 0)
236  throw std::runtime_error("Unknown element size");
237 
238  vertBuffer.name = name;
239  vertBuffer.elementCount = elementCount;
240  vertBuffer.elementSize = elementSize;
241  vertBuffer.values.resize(elementCount * elementSize);
242 
243  for(size_t i = 0; i < elementCount * elementSize; ++i)
244  {
245  stream.Read(vertBuffer.values[i]);
246  }
247 
248  ReadMarker(storage, kEndMarker);
249  }
250 
251  uint32_t unknown7 = 0;
252  uint32_t indexBufferCount = 0;
253  stream.Read(unknown7);
254  stream.Read(indexBufferCount);
255 
256  mesh.indexBuffers.resize(indexBufferCount);
257  for(unsigned int i = 0; i < indexBufferCount; ++i)
258  {
259  IndexBuffer& indexBuffer = mesh.indexBuffers[i];
260 
261  uint32_t unknown8, unknown9, operation, unknown11, indexCount;
262  stream.Read(unknown8);
263  stream.Read(unknown9);
264  stream.Read(operation);
265  stream.Read(unknown11);
266  stream.Read(indexCount);
267 
268  switch(operation)
269  {
270  case kTriangles:
271  indexBuffer.operation = IndexBuffer::kTriangles;
272  break;
273 
274  case kTriangleStrips:
275  indexBuffer.operation = IndexBuffer::kTriangleStrip;
276  break;
277 
278  default:
279  indexBuffer.operation = IndexBuffer::kUnknown;
280  }
281 
282  indexBuffer.indices.resize(indexCount);
283 
284  for(size_t i = 0; i < indexCount; ++i)
285  stream.Read(indexBuffer.indices[i]);
286 
287  ReadMarker(storage, kEndMarker);
288  }
289  ReadMarker(storage, kEndMarker);
290 }

294 {

296 
297  uint32_t unknown1 = 0; // always 0

299  ReadString(storage, program.source);
300 
301  uint32_t unknown2 = 0, unknown3 = 0, unknown4 = 0;
302  stream.Read(unknown2); // always 1
303  stream.Read(unknown3); // fp30: 0x11; vp30: 5
304  stream.Read(unknown4); // always 2
305  ReadString(storage, program.entryFunction);
306 
307  uint32_t marker = 0;
308  stream.Read(marker);
309  if(marker != kEndMarker)
310  {
311  uint32_t unknown5 = 0; // param set count? always 1
312  uint32_t unknown6 = 0; // 0x42000000
313  uint32_t unknown7 = 0; // 0
314  stream.Read(unknown5);
315  stream.Read(unknown6);
316  stream.Read(unknown7);
317  std::string paramSetName;
318  ReadString(storage, paramSetName);
319  uint32_t unknown8 = 0; // 1
320  uint32_t uniformCount = 0;
321  stream.Read(unknown8);
322  stream.Read(uniformCount);
323  for(unsigned int i = 0; i < uniformCount; ++i)
324  {
325  uint32_t type = 0;
326  uint32_t unknown9 = 0; // 0
327  std::string uniformName;
328 
329  stream.Read(type);

331  ReadString(storage, uniformName);
332

359

362  stream.Read(unknown11);
363  stream.Read(unknown12);
364  stream.Read(unknown13);
365  stream.Read(unknown14);
366  stream.Read(unknown15);
367  stream.Read(unknown16);
368  }
369  else if(type == kUniformWorldEyePos)
370  {
371  uint32_t unknown10 = 0; // 1
372  uint32_t unknown11 = 0; // 4
373  uint32_t unknown12 = 0; // 2

375 
376  stream.Read(unknown10);

380  }

383  uint32_t unknown10 = 0; // 1
