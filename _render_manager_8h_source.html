148 template<class TFileServer, class TTaskQueue>
149 bool RenderManagerT<TFileServer, TTaskQueue>::LoadProgramFile(const Blob& fileContents)
150 {
151  try
152  {
153  char* fileBegin = const_cast<char*>(static_cast<const char*>(fileContents.GetData()));
154  programgenerator::ProgramFile progFile(fileBegin, fileBegin + fileContents.GetSize());
155 
156  for(auto& it: progFile.GetVariables())
157  {
158  mProgramGenerator.AddVariable(it);
159  }
160 
161  for(auto& it2: progFile.GetFunctions())
162  {
163  mProgramGenerator.AddFunction(it2);
164  }
165  }
166  catch(const std::exception& e)
167  {
168  LOG(ERROR) << "Error loading program file: " << e.what();
169  return false;
170  }
171 
172  return true;
173 }
