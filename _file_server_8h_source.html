84  template<class TQueue>
85  void AddPackageFile(const std::string& path, TQueue& backgroundQueue)
86  {
87  try
88  {
89  const size_t headerSize = sizeof(PackageFile);
90  Blob headerData = SyncFileLoad(path.c_str(), mFileLoader, backgroundQueue, 0, headerSize);
91  const PackageFile* header = static_cast<const PackageFile*>(headerData.GetData());
92  if(header->magic != PackageFile::kMagic)
93  throw std::runtime_error("File is not a package file.");
94  const size_t entriesSize = header->count * sizeof(PackageFile::Entry);
95  Blob entryData = SyncFileLoad(path.c_str(), mFileLoader, backgroundQueue, headerSize, entriesSize);
96  const PackageFile::Entry* entriesBegin = static_cast<const PackageFile::Entry*>(entryData.GetData());
97  mPackageFileDirectory.Populate(path, entriesBegin, entriesBegin + header->count);
98  LOG(INFO) << "Package file " + path + " loaded. " << header->count << " files.";
99  }
100  catch(std::exception& e)
101  {
102  throw std::runtime_error("Loading of package file " + path + " failed: " + e.what());
103  }
104  }

163  void GetFileLocation(Hash file, char outPath[kMaxPathLength], size_t& outOffset, size_t& outSize) const
164  {
165  if(const char* path = mDirectoryContents.FindString(file))
166  {
167  strncpy(outPath, mRoot.c_str(), kMaxPathLength);
168  strncpy(outPath + mRoot.size(), path, kMaxPathLength - mRoot.size());
169  outOffset = 0;
170  outSize = 0;
171  }
172  else if(auto entry = mPackageFileDirectory.FindEntry(file))
173  {
174  strncpy(outPath, entry->file.c_str(), kMaxPathLength);
175  outOffset = entry->offset;
176  outSize = entry->size;
177  }
178  else
179  throw std::runtime_error("File not found");
180  }
