75  constexpr uint32_t Hash2(const char* str, size_t length, uint32_t state)
76  {
77  return
78  length == 0 ? state :
79  length == 1 ? Mix2(str[0]) ^ state :
80  length == 2 ? Mix2((str[1] << 8) | str[0]) ^ state :
81  length == 3 ? Mix2((str[2] << 16) | (str[1] << 8) | str[0]) ^ state :
82  /*else*/ Hash2(str + 4, length - 4, Mix(ToInt32(str), state));
83  }
