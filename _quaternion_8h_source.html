134 inline Quaternion Quaternion::operator*(const Quaternion& inQ) const
135 {
136  float w = W() * inQ.W() - X() * inQ.X() - Y() * inQ.Y() - Z() * inQ.Z();
137  float x = W() * inQ.X() + X() * inQ.W() + Y() * inQ.Z() - Z() * inQ.Y();
138  float y = W() * inQ.Y() + Y() * inQ.W() + Z() * inQ.X() - X() * inQ.Z();
139  float z = W() * inQ.Z() + Z() * inQ.W() + X() * inQ.Y() - Y() * inQ.X();
140  return Quaternion(w, x, y, z);
