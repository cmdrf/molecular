48 RenderCmdSink::Texture* Fbm3DTexture(RenderCmdSink& renderer, TFileServer& fileServer, unsigned int extent)
49 {
50  auto fbmVertFile = fileServer.Open("nvparticles/fbm.vert");
51  auto fbmFragFile = fileServer.Open("nvparticles/fbm.frag");
52  RenderCmdSink::Program* fbmProgram = renderer.CreateProgram();
53  auto fbmVertSrc = StringUtils::FromStorage(fbmVertFile);
54  auto fbmFragSrc = StringUtils::FromStorage(fbmFragFile);
55  fbmProgram->Store(fbmVertSrc, fbmFragSrc);
56 
57  RenderCmdSink::Texture* out = renderer.CreateTexture();
58  out->Store(extent, extent, extent, NULL, PF_RGBA_FLOAT16);
59  out->SetParameter(RenderCmdSink::Texture::kMinFilter, RenderCmdSink::Texture::kLinear);
60  out->SetParameter(RenderCmdSink::Texture::kMagFilter, RenderCmdSink::Texture::kLinear);
61 
62  RenderCmdSink::RenderTarget* framebuffer = renderer.CreateRenderTarget();
63  RenderCmdSink::VertexBuffer* quadVertexBuffer = renderer.CreateVertexBuffer();
64 
65  const float position[] = {
66  -1.0f, -1.0f,
67  1.0f, -1.0f,
68  -1.0f, 1.0f,
69  1.0f, 1.0f,
70  };
71 
72  quadVertexBuffer->Store(position, sizeof(float) * 8);
73  fbmProgram->SetAttribute("attr_position"_H, quadVertexBuffer, 2, VertexAttributeInfo::kFloat);
74 
75  for(unsigned int n = 0; n < extent; n++)
76  {
77  const float layer = n/(float)extent;
78 
79  framebuffer->AttachColorBuffer(out, 0, n);
80  renderer.SetTarget(framebuffer);
81 
82  renderer.UseProgram(fbmProgram);
83  fbmProgram->SetUniform("u_Layer"_H, &layer);
84 
85  renderer.Draw(IndexBufferInfo::Mode::kTriangleStrip, 4);
86  }
87  renderer.SetTarget(nullptr);
88  renderer.DestroyVertexBuffer(quadVertexBuffer);
89  renderer.DestroyRenderTarget(framebuffer);
90  renderer.DestroyProgram(fbmProgram);
91 
92  return out;
93 }
