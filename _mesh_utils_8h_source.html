119 template<class Attribute0, class Attribute1>
120 void SeparateToUnifiedIndices(
121  size_t numIndices,
122  const uint32_t indices0[],
123  const uint32_t indices1[],
124  size_t numAttributes0, const Attribute0 attributes0[],
125  size_t numAttributes1, const Attribute1 attributes1[],
126  std::vector<uint32_t>& outIndices,
127  std::vector<Attribute0>& outAttributes0,
128  std::vector<Attribute1>& outAttributes1)
129 {
130  if(indices0 && indices1)
131  {
132  std::unordered_map<uint64_t, uint32_t> vertexMap;
133 
134  for(size_t i = 0; i < numIndices; ++i)
135  {
136  uint32_t index0 = indices0[i];
137  uint32_t index1 = indices1[i];
138 
139  const uint64_t combinedIndex = (static_cast<uint64_t>(index0) << 32) | index1;
140  auto it = vertexMap.find(combinedIndex);
141 
142  if(it == vertexMap.end())
143  {
144  uint32_t outIndex = outAttributes0.size();
145  vertexMap[combinedIndex] = outIndex;
146  outIndices.push_back(outIndex);
147 
148  assert(numAttributes0 > index0);
149  outAttributes0.push_back(attributes0[index0]);
150 
151  assert(numAttributes1 > index1);
152  outAttributes1.push_back(attributes1[index1]);
153  }
154  else
155  outIndices.push_back(it->second);
156  }
157  }
158  else if(indices0)
159  {
160  outIndices.insert(outIndices.end(), indices0, indices0 + numIndices);
161  outAttributes0.insert(outAttributes0.end(), attributes0, attributes0 + numAttributes0);
162  }
163  else if(indices1)
164  {
165  outIndices.insert(outIndices.end(), indices1, indices1 + numIndices);
166  outAttributes1.insert(outAttributes1.end(), attributes1, attributes1 + numAttributes1);
167  }
168 }
