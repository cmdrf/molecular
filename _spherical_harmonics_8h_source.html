71 template<int numBands, typename PolarFunction>
72 Vector<numBands * numBands, double> ProjectPolarFunction(PolarFunction func, const std::vector<Sample<numBands>>& samples)
73 {
74  Vector<numBands * numBands, double> result;
75  // for each sample
76  for(auto& sample: samples)
77  result += sample.coeff * func(sample.theta, sample.phi);
78 
79  // divide the result by weight and number of samples
80  const double weight = 4.0 * Math::kPi_d;
81  result *= weight / samples.size();
82  return result;
83 }
