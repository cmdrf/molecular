74  Matrix3(const Quaternion& quaternion)
75  {
76  ValueType xx = quaternion.X() * quaternion.X();
77  ValueType xy = quaternion.X() * quaternion.Y();
78  ValueType xz = quaternion.X() * quaternion.Z();
79  ValueType xw = quaternion.X() * quaternion.W();
80  ValueType yy = quaternion.Y() * quaternion.Y();
81  ValueType yz = quaternion.Y() * quaternion.Z();
82  ValueType yw = quaternion.Y() * quaternion.W();
83  ValueType zz = quaternion.Z() * quaternion.Z();
84  ValueType zw = quaternion.Z() * quaternion.W();
85  m[0][0] = 1 - 2 * ( yy + zz );
86  m[0][1] = 2 * ( xy - zw );
87  m[0][2] = 2 * ( xz + yw );
88  m[1][0] = 2 * ( xy + zw );
89  m[1][1] = 1 - 2 * ( xx + zz );
90  m[1][2] = 2 * ( yz - xw );
91  m[2][0] = 2 * ( xz - yw );
92  m[2][1] = 2 * ( yz + xw );
93  m[2][2] = 1 - 2 * ( xx + yy );
94  }

122  Quaternion ToQuaternion() const
123  {
124  ValueType trace = m[0][0] + m[1][1] + m[2][2];
125  ValueType w, x, y, z;
126  if( trace > 0 )
127  {
128  float s = 0.5f / std::sqrt(trace + 1.0f);
129  w = 0.25f / s;
130  x = ( m[2][1] - m[1][2] ) * s;
131  y = ( m[0][2] - m[2][0] ) * s;
132  z = ( m[1][0] - m[0][1] ) * s;
133  }
134  else
135  {
136  if(m[0][0] > m[1][1] && m[0][0] > m[2][2])
137  {
138  float s = 2.0f * std::sqrt(1.0f + m[0][0] - m[1][1] - m[2][2]);
139  w = (m[2][1] - m[1][2] ) / s;
140  x = 0.25f * s;
141  y = (m[0][1] + m[1][0] ) / s;
142  z = (m[0][2] + m[2][0] ) / s;
143  }
144  else if(m[1][1] > m[2][2])
145  {
146  float s = 2.0f * std::sqrt( 1.0f + m[1][1] - m[0][0] - m[2][2]);
147  w = (m[0][2] - m[2][0] ) / s;
148  x = (m[0][1] + m[1][0] ) / s;
149  y = 0.25f * s;
150  z = (m[1][2] + m[2][1] ) / s;
151  }
152  else
153  {
154  ValueType s = 2.0f * std::sqrt( 1.0f + m[2][2] - m[0][0] - m[1][1] );
155  w = (m[1][0] - m[0][1] ) / s;
156  x = (m[0][2] + m[2][0] ) / s;
157  y = (m[1][2] + m[2][1] ) / s;
158  z = 0.25f * s;
159  }
