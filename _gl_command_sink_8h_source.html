362  {
363  gl.BindTransformFeedback(gl.TRANSFORM_FEEDBACK, mFeedback);
364  CheckError("glBindTransformFeedback", __LINE__, __FILE__);
365  gl.BeginTransformFeedback(ToGlEnum(mode));
366  CheckError("glBeginTransformFeedback", __LINE__, __FILE__);
367  if(mQuery)

369  gl.BeginQuery(gl.TRANSFORM_FEEDBACK_PRIMITIVES_WRITTEN, mQuery);
370  CheckError("glBeginQuery", __LINE__, __FILE__);
371  }
372  }
373 
374  void End()
375  {
376  gl.EndTransformFeedback();
377  CheckError("glEndTransformFeedback", __LINE__, __FILE__);
378  if(mQuery)

380  gl.EndQuery(gl.TRANSFORM_FEEDBACK_PRIMITIVES_WRITTEN);
381  CheckError("glEndQuery", __LINE__, __FILE__);
382  }
383  }
384 
385 private:
386  TransformFeedback(GLuint name) : mFeedback(name)
387  {
388  if(!gl.HasDrawTransformFeedback())
389  gl.GenQueries(1, &mQuery);
390  }
