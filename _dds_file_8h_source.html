235 template<class Storage>
236 void DdsFile::ReadHeader(Storage& storage)
237 {
238  LittleEndianReadStream<Storage> stream(storage);
239 
240  uint32_t magic;
241  stream.Read(magic);
242  if(magic != 0x20534444)
243  throw std::runtime_error("Data is no DDS file");
244 
245 #if __LITTLE_ENDIAN__
246  storage.Read(&mHeader, 124);
247 #else
248  stream.Read(mHeader.size);
249  stream.Read(mHeader.flags);
250  stream.Read(mHeader.height);
251  stream.Read(mHeader.width);
252  stream.Read(mHeader.pitchOrLinearSize);
253  stream.Read(mHeader.depth);
254  stream.Read(mHeader.mipMapCount);
255  for(int i = 0; i < 11; ++i)
256  stream.Read(mHeader.reserved1[i]);
257  stream.Read(mHeader.pixelFormat.size);
258  stream.Read(mHeader.pixelFormat.flags);
259  stream.Read(mHeader.pixelFormat.fourCc);
260  stream.Read(mHeader.pixelFormat.rgbBitCount);
261  stream.Read(mHeader.pixelFormat.rBitMask);
262  stream.Read(mHeader.pixelFormat.gBitMask);
263  stream.Read(mHeader.pixelFormat.bBitMask);
264  stream.Read(mHeader.pixelFormat.aBitMask);
265  stream.Read(mHeader.caps);
266  stream.Read(mHeader.caps2);
267  stream.Read(mHeader.caps3);
268  stream.Read(mHeader.caps4);
269  stream.Read(mHeader.reserved2);
270 #endif
271  if(mHeader.size != 124)
272  throw std::runtime_error("Invalid header data");
273 
274  if(!(mHeader.flags & (Header::kCaps | Header::kPixelformat | Header::kWidth | Header::kHeight)))
275  throw std::runtime_error("Insufficient header data");
276 
277  if(!(mHeader.caps & Header::kTexture))
278  throw std::runtime_error("No texture data present");
279 
280  mFacesMipmapLevels.resize(mNumFaces * mHeader.mipMapCount);
281 }
282 
283 }
