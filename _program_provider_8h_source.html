67 template<class Iterator, typename ArraySizeFunc>
68 RenderCmdSink::Program* ProgramProvider::GetProgram(Iterator varsBegin, Iterator varsEnd, ArraySizeFunc&& arraySizeFunc)
69 {
70  // Calculate hash of all variables:
71  Hash hash = CalculateHash(varsBegin, varsEnd);
72 
73  // Search for cached program:
74  auto it = mGeneratedPrograms.find(hash);
75  if(it != mGeneratedPrograms.end())
76  return it->second; // Cached program found, return it
77  else
78  {
79  // Program not cached, generate it
80  for(auto it = varsBegin; it != varsEnd; ++it)
81  {
82  if(mArraySizes.count(*it) == 0)
83  mArraySizes[*it] = arraySizeFunc(*it);
84  }
85  programgenerator::ProgramGenerator::ProgramText text = mGenerator.GenerateProgram(varsBegin, varsEnd, mArraySizes);
86  RenderCmdSink::Program* program = mRenderer.CreateProgram();
87  program->Store(text.vertexShader, text.fragmentShader);
88  mGeneratedPrograms[hash] = program;
89  return program;
90  }
91 }
