38 template<class TRenderManager, class TZoneKey>
39 void PortalZoneScene<TRenderManager, TZoneKey>::Execute()
40 {
41  const Uniform<Matrix4>* projMatrix = GetVariable<Uniform<Matrix4> >("projectionMatrix"_H);
42  const Uniform<Matrix4>* viewMatrix = GetVariable<Uniform<Matrix4> >("viewMatrix"_H);
43 
44  if(!projMatrix || !viewMatrix)
45  return;
46 
47  Matrix4 viewProjectionMatrix = **projMatrix * **viewMatrix;
48  util::Frustum frustum(viewProjectionMatrix);
49  auto planes = frustum.GetPlanes();
50 
51  Vector3 cameraPos = Matrix4((**viewMatrix).Inverse()).GetTranslation();
52  mCurrentViewZone = FindZone(cameraPos, mCurrentViewZone);
53  if(mCurrentViewZone != mZones.end())
54  {
55  auto& zone = mCurrentViewZone->second;
56  mDebugMesh.DrawAxisAlignedBox(zone.bounds.GetMin(), zone.bounds.GetMax(), Vector3(0, 1, 0));
57  DrawZone(zone, planes.begin(), planes.end(), cameraPos);
58  }
59 
60  for(auto func: mFunctions)
61  {
62  auto bounds = func->GetBounds(); // TODO: Cache
63  if(frustum.Check(bounds) != util::Plane::kOutside)
64  func->Execute();
65  }
66 
67  // Debug:
68  Binding<Uniform<Matrix4>> modelMatrix("modelMatrix"_H, this);
69  **modelMatrix = Matrix4::Identity();
70  mDebugMesh.Execute();
71 }

171 inline bool PortalFacingToViewer(const Vector3 portalPoints[3], const Vector3& viewPos)
172 {
173  Vector3 kj = portalPoints[0] - portalPoints[1];
174  Vector3 lj = portalPoints[0] - portalPoints[2];
175  Vector3 normal = kj.CrossProduct(lj);
176  return normal.DotProduct(portalPoints[0] - viewPos) < 0;
177 }

181 void PortalZoneScene<TRenderManager, TZoneKey>::DrawZone(Zone& zone, PlaneIterator planesBegin, PlaneIterator planesEnd, const Vector3& viewPos)
182 {
183  if(zone.sky)
184  zone.sky->Execute();
185  for(auto& func: zone.functions)
186  {
187  auto bounds = func->GetBounds(); // TODO: Cache
188  if(PlaneSet::CheckIntersection(planesBegin, planesEnd, bounds) != Plane::kOutside)
189  func->Execute();
190  }
191 
192  for(unsigned int i = 0; i < zone.numPortals; ++i)
193  {
194  auto& portal = zone.portals[i];
195  assert(portal.numPoints >= 3);
196  if(!PortalFacingToViewer(portal.points, viewPos))
197  continue;
198 
199  auto nextZoneIt = mZones.find(portal.destinationZone);
200  if(nextZoneIt == mZones.end())
201  continue;
202  auto& nextZone = nextZoneIt->second;
203 
204  std::array<Vector3, 128> clippedPortal;
205  auto clippedPortalEnd = util::PlaneSet::ClipPolygon(planesBegin, planesEnd, portal.points, portal.points + portal.numPoints, clippedPortal.begin());
206  auto numPoints = clippedPortalEnd - clippedPortal.begin();
207  if(numPoints == 0 || numPoints >= clippedPortal.size() - 2)
208  continue;
209 
210  for(unsigned int i = 0; i < numPoints; i++)
211  mDebugMesh.DrawLine(clippedPortal[i], clippedPortal[(i + 1) % numPoints], Vector3(1, 0, 0));
212 
213  std::array<util::Plane, 128> clippedPlanes;
214  auto clippedPlanesBegin = std::copy(planesBegin, planesBegin + 2, clippedPlanes.begin()); // Copy near and far plane TODO: Use portal as near plane
215  auto clippedPlanesEnd = util::PlaneSet::CreateCullingPlanes(viewPos, clippedPortal.begin(), clippedPortalEnd, clippedPlanesBegin);
216 
217  assert(clippedPlanesEnd - clippedPlanes.begin() == numPoints + 2);
218  DrawZone(nextZone, clippedPlanes.begin(), clippedPlanesEnd, viewPos);
219  }
220 }
