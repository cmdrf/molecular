71 void CpuParticleSystem<TRenderManager>::Execute()
72 {
73  bool shadowPass = !mScoping["fragmentColor"_H];
74 
75  if((shadowPass && mCastShadows) || (!shadowPass && !mCastShadows))
76  {
77  if(mTaskQueue.IsFinished(mSimulationFinishFlag))
78  {
79  // TODO: Keep buffer size constant
80  if(!mParticles.empty())
81  mVertexBuffer->Store(mParticles.data(), mParticles.size() * sizeof(ParticleVec), true);
82  mNumParticles = mParticles.size();
83  mSimulationTaskEmitters = mEmitters;
84  auto simulate = [this]()
85  {
86  RemoveOld(mMaxParticleAge);
87  const float t = 1.0f / 60.0f; // TODO
88  Transition(t, mTransitionRandomSigma, mAdditive);
89  Emission(t, mEmissionRandomSigma, mEmitsPerSecond);
90  };
91  mTaskQueue.EnqueueTask(simulate, mSimulationFinishFlag);
92  }
93  }
94 
95  if(!shadowPass || mCastShadows)
96  {
97  Binding<Attribute> position("vertexPositionAttr"_H, this);
98  Binding<Attribute> particleAge("particleAge"_H, this);
99  Binding<Uniform<float>> particleMaxAge("particleMaxAge"_H, this);
100  Binding<Uniform<Vector3>> color("diffuseColor"_H, this);
101  Binding<Uniform<Matrix4>> modelMatrix("modelMatrix"_H, this);
102  Binding<Uniform<float>> pointWorldSize("pointWorldSize"_H, this);
103  Binding<Output> glPointSizeOutput("gl_PointSize"_H, this);
104 
105  *position = Attribute(mVertexBuffer, mPositionInfo);
106  *particleAge = Attribute(mVertexBuffer, mAgeInfo);
107  **particleMaxAge = mMaxParticleAge;
108  **color = mColor;

151 template<class TRenderManager>
152 void CpuParticleSystem<TRenderManager>::Transition(float t, ParticleVec randomSigma, ParticleVec additive)
153 {
154  for(auto& particle: mParticles)
155  {
156  particle.head(3) += particle.segment(kIndexVelX, 3) * t;
157  particle += additive * t;
158  particle(kIndexAge) += t;
159  particle += ParticleVec::Random().cwiseProduct(randomSigma);
160  }
161 }

163 template<class TRenderManager>
164 void CpuParticleSystem<TRenderManager>::Emission(float t, ParticleVec randomSigma, float emitsPerSecond)
165 {
166  int emissionsPerEmitter = emitsPerSecond * t;
167  for(auto& emitter: mSimulationTaskEmitters)
168  {
169  for(int i = 0; i < emissionsPerEmitter; ++i)
170  {
171  if(mParticles.size() >= kMaxParticles)
172  return;
173 
174  mParticles.push_back(emitter + ParticleVec::Random().cwiseProduct(randomSigma));
175  }
176  }
177 }
