107 template<class TQueue>
108 dispatch_source_t GcdAsyncFileLoader::SetupSource(const char* path, std::function<void (Blob&)> handler, TQueue& handlerQueue, typename TQueue::FinishFlag* finishFlag, size_t offset, size_t length)
109 {
110  int fd = open(path, O_RDONLY | O_NONBLOCK);
111  if(fd == -1)
112  throw std::runtime_error("GcdAsyncFileLoader: Could not open file " + std::string(path));
113 
114  if(length == 0)
115  {
116  struct stat output;
117  if(fstat(fd, &output) != 0)
118  throw std::runtime_error("GcdAsyncFileLoader: fstat on " + std::string(path) + " failed");
119  if(output.st_size == 0)
120  throw std::runtime_error("GcdAsyncFileLoader: Empty file " + std::string(path));
121  length = output.st_size - offset;
122  }
123 
124  if(offset != 0)
125  lseek(fd, offset, SEEK_SET);
126  dispatch_source_t dispatchSource = dispatch_source_create(DISPATCH_SOURCE_TYPE_READ, fd, 0, mQueue.mQueue);
127 
128  if(!dispatchSource)
129  {
130  close(fd);
131  throw std::runtime_error("GcdAsyncFileLoader: Could not create dispatch source");
132  }
133 
134  ReadContext<TQueue>* c = new ReadContext<TQueue>
135  {
136  Blob(length),
137  handler,
138  mQueue,
139  handlerQueue,
140  finishFlag,
141  fd,
142  0,
143  dispatchSource
144  };
145  dispatch_set_context(dispatchSource, c);
146  dispatch_source_set_event_handler_f(dispatchSource, ReadEventHandler<TQueue>);
147 
148  return dispatchSource;
149 }

173 template<class TQueue>
174 void GcdAsyncFileLoader::ReadEventHandler(void* context)
175 {
176  ReadContext<TQueue>* readContext = static_cast<ReadContext<TQueue>*>(context);
177  uint8_t* destination = readContext->blob.GetBytes() + readContext->completed;
178  size_t bytesToRead = readContext->blob.GetSize() - readContext->completed;
179  ssize_t result = read(readContext->fileDescriptor, destination, bytesToRead);
180 // std::cerr << "Read " << result << std::endl;
181  readContext->completed += result;
182  if(readContext->completed >= readContext->blob.GetSize())
183  dispatch_source_cancel(readContext->dispatchSource);
184 }
